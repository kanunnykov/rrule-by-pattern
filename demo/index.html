    freq: RRule.DAILY,
    dtstart: new Date(Date.UTC(2026, 0, 1, 6)),

const builder = new ReadableRRuleSetBuilder(
    rruleSet,
    template,

const newRRuleSet = builder.build()

console.log(newRRuleSet.all())
